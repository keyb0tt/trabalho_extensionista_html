Algoritmo "Cálculo de Áreas"

Var
opcao, n1, n2, i: inteiro

Inicio
i <- 0

Escreval("Olá, seja bem-vindo ao sistema de cálculo de áreas!")
Enquanto (i = 0) faca
  Escreval("Cálculo de área de: ")
  Escreval("Círculo: 1")
  Escreval("Paralelepípedo: 2")
  Escreval("Triângulo: 3")
  Escreval("Quadrado: 4")
  Escreval("Pentágono: 5")
  Leia(opcao)

  Se (opcao = 1) entao
     escreval("Digite o raio do círculo em cm")
     leia(n1)
     Escreval("A área de um círculo de raio", n1, " é ", ((n1 * n1) * 3.1415926535), " cm²")
  fimse

  Se (opcao = 2) entao
     escreval("Digite a largura do paralelepípedo em cm")
     leia(n1)
     escreval("Digite o comprimento do paralelepípedo em cm")
     leia(n2)
     Escreval("A área do paralelepípedo de largura", n1, " e comprimento", n2, " é ", (n1 * n2), " cm²")
  fimse

  Se (opcao = 3) entao
     escreval("Digite a altura do triângulo em cm")
     leia(n1)
     escreval("Digite a base do triângulo em cm")
     leia(n2)
     Escreval("A área do triângulo de altura", n1, " e base", n2, " é ", ((n1 * n2)/ 2), " cm²")
  fimse

  Se (opcao = 4) entao
     escreval("Digite o lado do quadrado em cm")
     leia(n1)
     Escreval("A área do quadrado de lado", n1, " é ", (n1 * n1), " cm²")
  fimse

  Se (opcao = 5) entao
     escreval("Digite o lado do pentágono regular em cm")
     leia(n1)
     Escreval("A área do pentágono regular de lado", n1, " é ", (5 * n1 * n1) / (4 * 0.7265), " cm²")
  fimse

  Escreval("Deseja sair?")
  Escreval("Sim: 1")
  Escreval("Não: 2")
  leia(i)
  se (i = 1) entao
     i <- 1
  senao
     i <- 0
  fimse
fimenquanto
